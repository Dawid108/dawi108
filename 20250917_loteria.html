Kod w Pythonie:

import random

def generuj_losowania(n):
    wyniki = []
    for _ in range(n):
        zestaw = random.sample(range(1, 50), 6)
        zestaw.sort()
        wyniki.append(zestaw)
    return wyniki

def wypisz_losowania(wyniki):
    for i, zestaw in enumerate(wyniki, start=1):
        print(f"Losowanie {i}: {', '.join(map(str, zestaw))}")

def policz_wystapienia(wyniki):
    wystapienia = [0] * 50
    for zestaw in wyniki:
        for liczba in zestaw:
            wystapienia[liczba] += 1
    return wystapienia

def wypisz_wystapienia(wystapienia):
    for liczba in range(1, 50):
        print(f"Wystąpienia liczby {liczba}: {wystapienia[liczba]}")

def main():
    print("Program: Loteria liczbowa (6 z 49)")
    n = int(input("Ile zestawów wylosować? "))
    wyniki = generuj_losowania(n)
    print("\nZestawy wylosowanych liczb:")
    wypisz_losowania(wyniki)
    wystapienia = policz_wystapienia(wyniki)
    print("\nStatystyka wystąpień:")
    wypisz_wystapienia(wystapienia)

if __name__ == "__main__":
    main()
