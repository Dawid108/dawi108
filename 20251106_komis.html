Próg sprawności: < 180000 km
python · workshop.py
// workshop.py

class Workshop:
    def __init__(self):
        self._przebieg = 0  # prywatny atrybut: przebieg

    @property
    def przebieg(self):
        return self._przebieg

    @przebieg.setter
    def przebieg(self, a):
        # setter pyta o przebieg i sprawdza, czy kupujemy pojazd
        self._przebieg = int(input("Podaj przebieg w km: "))

        if self._przebieg > 180000:
            print("Dziękujemy, twój pojazd ma za duży lub niepoprawny przebieg, więc go nie kupimy.")
        else:
            print("Dziękujemy, kupimy twój pojazd!")
